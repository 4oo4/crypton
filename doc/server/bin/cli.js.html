'use strict';

var fs = require('fs');
var program = require('commander');

program
  .version('0.0.2')
  .option('-c, --config [file]', 'Specify a custom configuration file [default config]')
  .parse(process.argv);

process.configFile = program.config;

if (require.main === module) {
  var app = require('../app');
  app.start();
}
