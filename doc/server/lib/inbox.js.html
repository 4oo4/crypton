var app = require('../app');
var db = app.datastore;

var Inbox = module.exports = function Inbox (id) {
  this.accountId = id;
};

Inbox.prototype.getAllMessages = function (callback) {
  db.getAllMessages(this.accountId, function (err, messages) {
    callback(err, messages);
  });
};
